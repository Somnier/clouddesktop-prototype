# 云桌面业务对象关系图

> 过滤了跨域的虚线依赖和业务动作流转，只保留业务对象的**组合(*--)** 与**聚合(o--)** 关系，聚焦于业务对象数据属性。让人第一眼看清：整个业务盘子里到底有哪几个实体，实体下面挂着什么数据，服务归属于谁。

```mermaid
classDiagram
    %% ================= 服务器域 =================
    namespace 服务器域 {
        class 服务器 {
            IP地址
            域名
        }
        class 服务器运行状态 {
            CPU使用率
            内存使用率
            存储使用率
        }
        class 授权许可证书 {
            授权点数
            有效期限
        }
        class 告警记录 {
            告警类型
            异常来源设备(MAC或服务器)
            溯源位置(教室与座位号)
            处理状态
        }
        class 管理员账号 {
            账号角色
            管理范围
        }
        class 云端基础镜像 {
            OS类型
        }
        class 服务器端快照 {
            桌面名称
            增量数据
        }
        class 服务器端终端登记 {
            已登记的MAC地址
            绑定的教室与座位
            终端联机状态
            终端当前CPU使用率
            终端当前内存使用率
        }
    }

    %% ================= 终端服务域 =================
    namespace 终端服务域 {
        class 终端 {
            物理MAC地址列表
            当前工作MAC地址
            硬盘SN码
            CPU型号
            内存型号
            硬盘型号
            外设连接状态
        }
        class 终端配置 {
            分配的IP地址
            计算机名
            座位号
            关联服务器地址
        }
        class 终端运行状态 {
            网络联机状态
            CPU使用率
            内存占有率
            磁盘读写负载
        }
        class 逻辑教室 {
            教室名称
            教室布局图
        }
        class 教室终端清单 {
            记录的终端MAC地址
            记录的终端IP地址
            记录的计算机名
            记录的座位号
            记录的硬盘SN码
        }
        class 安装U盘 {
            存储容量
        }
        class 本地基础镜像 {
            OS类型
        }
        class 本地快照 {
            还原模式
        }
        class 终端底层服务 {
            批量部署桌面()
            克隆与还原桌面()
            修改桌面启动项()
            导出教室终端清单()
        }
        class 操作日志 {
            操作人
            时间
            动作记录
            执行结果
        }
    }

    %% ================= 终端引导启动域 =================
    namespace 终端引导启动域 {
        class 启动菜单 {
            可用启动项列表
            默认启动项
        }
    }

    %% ================= 终端桌面域 =================
    namespace 终端桌面域 {
        class 桌面 {
            桌面名称
        }
        class 云桌面客户端服务 {
            同步本地桌面资产()
            接收关机与重启指令()
        }
        class 教室管控服务 {
            发送一键开机指令()
            发送一键关机与重启指令()
        }
    }

    %% ================= 核心结构连线 (极简版) =================
    服务器 "1" *-- "1" 服务器运行状态
    服务器 "1" *-- "n" 告警记录
    服务器 "1" o-- "1" 授权许可证书
    服务器 "1" *-- "n" 管理员账号
    服务器 "1" *-- "n" 服务器端终端登记
    服务器 "1" o-- "n" 云端基础镜像
    云端基础镜像 "1" *-- "n" 服务器端快照

    服务器 "1" *-- "n" 逻辑教室
    逻辑教室 "1" *-- "n" 终端配置
    逻辑教室 "1" *-- "1" 教室终端清单

    终端 "1" *-- "1" 终端配置
    终端 "1" *-- "1" 终端运行状态

    终端底层服务 "1" *-- "n" 操作日志

    安装U盘 o-- 本地基础镜像
    本地基础镜像 "1" *-- "n" 本地快照

    启动菜单 "1" o-- "n" 本地快照
    本地快照 "1" *-- "1" 桌面

    桌面 "1" *-- "1" 云桌面客户端服务
    桌面 "1" o-- "1" 教室管控服务
```
