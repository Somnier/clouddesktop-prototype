# 云桌面业务对象关系图 v5

> **版本说明 (v5)**：
> 1. **深度洞察“逻辑教室”**：引入聚合关系（`o--`）代替组合关系（`*--`）连接教室与终端。这在架构上完美支撑了“同一个物理终端可以既属于小班教室，又属于合班大教室”的物理位置重叠场景。
> 2. **克制而精准的连线**：去掉了零碎的操作动作连线，保留了最核心的**资产流转**（镜像->快照衍生、本地<->服务器双向同步）和**逻辑映射**（清单转登记、登记映射物理机）。

```mermaid
classDiagram
    %% ================= 服务器域 =================
    namespace 服务器域 {
        class 服务器 {
            IP地址
            域名
        }
        class 服务器运行状态 {
            CPU使用率
            内存使用率
            存储使用率
        }
        class 授权许可证书 {
            授权点数
            有效期限
        }
        class 管理员账号 {
            账号角色
            管理范围
        }
        class 服务器端基础镜像 {
            OS名称
            OS类型
            OS大小
        }
        class 服务器端快照 {
            快照名称
            快照大小
        }
        class 服务器端终端登记 {
            已登记的终端MAC地址：多网卡
            终端绑定的逻辑教室与座位
            终端联机状态
        }
        class 服务器端告警记录 {
            告警类型
            异常来源设备：终端或服务器自身
            溯源位置：教室与座位号
            处理状态
        }
    }

    %% ================= 终端服务域 =================
    namespace 终端服务域 {
        class 终端 {
            物理MAC地址：多网卡
            当前工作MAC地址
            硬盘SN码
            CPU型号
            内存型号
            硬盘型号
            外设物理连接状态
        }
        class 终端配置 {
            分配的IP地址
            计算机名
            逻辑教室与座位号：多教室同终端
            关联服务器地址
        }
        class 终端运行状态 {
            与服务器联机状态
            CPU使用率
            内存使用率
            存储使用率
        }
        class 逻辑教室 {
            教室名称
            教室布局图：终端座位排布
        }
        class 教室终端清单 {
            记录的终端MAC地址
            记录的终端IP地址
            记录的计算机名
            记录的座位号
            记录的硬盘SN码
        }
        class 安装U盘 {
            云桌面裸机安装程序
        }
        class 本地基础镜像 {
            OS名称
            OS类型
            OS大小
        }
        class 本地快照 {
            快照名称
            快照大小
        }
        class 终端底层服务框架 {
            本地部署桌面()
            批量下发桌面()
            克隆与还原桌面()
            修改桌面启动项()
            导出教室终端清单()
        }
        class 操作日志 {
            操作人
            时间
            动作记录
            执行结果
        }
        class 本地告警记录 {
            告警类型
            处理状态
            与服务器同步处理状态()
        }
    }

    %% ================= 终端引导启动域 =================
    namespace 终端引导启动域 {
        class 启动菜单 {
            可用启动项列表
            默认启动项
            还原模式
        }
    }

    %% ================= 终端桌面域 =================
    namespace 终端桌面域 {
        class 桌面 {
            桌面名称
        }
        class 云桌面客户端服务 {
            与服务器同步本地桌面()
            接收关机与重启指令()
        }
        class 教室管控服务 {
            发送一键开机指令()
            发送一键关机与重启指令()
        }
    }

    %% ================= 核心实体结构与跨域流转关系 =================
    %% 服务器内部结构
    服务器 "1" *-- "1" 服务器运行状态
    服务器 "1" *-- "n" 服务器端告警记录
    服务器 "1" o-- "1" 授权许可证书
    服务器 "1" *-- "n" 管理员账号
    服务器 "1" *-- "n" 服务器端终端登记
    服务器 "1" o-- "n" 服务器端基础镜像
    服务器端基础镜像 "1" <|-- "n" 服务器端快照 : 派生

    %% 教室与终端的关联 (逻辑教室可重叠，采用聚合而非组合)
    服务器 "1" *-- "n" 逻辑教室
    逻辑教室 "1" o-- "n" 终端 : 包含
    逻辑教室 "1" *-- "1" 教室终端清单 : 导出时的存档数据

    %% 终端内部结构
    终端 "1" *-- "1" 终端配置 : 承载逻辑身份与网络配置
    终端 "1" *-- "1" 终端运行状态
    终端 "1" *-- "1" 终端底层服务框架
    终端 "1" ..> "1" 安装U盘 : 裸机安装
    终端底层服务框架 "1" ..> "1" 安装U盘 : 获取镜像
    终端底层服务框架 "1" ..> "n" 本地基础镜像 : 持有
    终端底层服务框架 "1" *-- "n" 操作日志
    终端底层服务框架 "1" *-- "n" 本地告警记录
    本地基础镜像 "1" <|-- "n" 本地快照 : 派生

    %% 资产关系
    安装U盘 o-- 本地基础镜像

    %% 跨域核心业务实体流转与映射
    教室终端清单 ..> 服务器端终端登记 : 数据载体(导入建档)
    服务器端终端登记 --> 终端 : 逻辑映射物理设备

    本地基础镜像 ..> 服务器端基础镜像 : 上传基础数据
    本地快照 <..> 服务器端快照 : 增量差异双向同步
    本地告警记录 <..> 服务器端告警记录 : 双向同步

    %% 引导与桌面层结构
    终端底层服务框架 ..> 启动菜单 : 维护与修改
    启动菜单 --> 本地快照 : 指向引导目标
    本地快照 --> 桌面 : 加载为运行环境

    桌面 ..> 终端配置 : 读取IP与计算机名以完成OS初始化
    桌面 "1" *-- "1" 云桌面客户端服务
    桌面 "1" o-- "1" 教室管控服务 : 教师机额外安装
```
